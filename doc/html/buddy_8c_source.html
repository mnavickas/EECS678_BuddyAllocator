166 void *buddy_alloc(int size)
167 {
168 
169  #if USE_DEBUG == 1
170  buddy_dump();
171  PDEBUG("Allocating Size of %dK\n", size/1024);
172  #endif
173  if( size <= 0 )
174  {
175  return NULL;
176  }
177  /*
178  1. Ascertain the free-block order which can satisfy the requested size.
179  The block order for size x is ceil ( log2 (x))
180  Example:
181  60k -> block-order = ceil ( log2 (60k)) =
182  ceil ( log2 (k x 2^5 x 2^10)) = order-16
183  */
184  const int REQ_ORDER = MAX( ceil( log2( size ) ), MIN_ORDER );
185 
186 
187  for(int i = REQ_ORDER; i <= MAX_ORDER; i++ )
188  {
189  /*
190  2. Iterate over the free-lists; starting from the order calculated in the
191  above step. If the free-list at the required order is not-empty, just remove
192  the first page from that list and return it to caller to satisfy the request
193  */
194  if( !list_empty(&free_area[i]) )
195  {
196  return buddy_split(i, REQ_ORDER);
197  }
198  else
199  {
200  /*
201  3. If the free-list at the required order is empty, find the first
202  non-empty free-list with order > required-order. Lets say that such a
203  list exists at order-i
204  */
205  }
206  }
207  /*
208  5. If a non-empty free-list is not found, this is an error
209  */
210  return NULL;
211 }

244 void buddy_free(void *addr)
245 {
246 
247  char* address = (char*)addr;
248  /*
249  1. Calculate the address of the buddy
250  */
251  int index = ADDR_TO_PAGE ( address );
252  int order = g_pages[index].order;
253 
254  /*
255  2. If the buddy is free, merge the two blocks i.e. remove the buddy from its
256  free-list, update the order of the page-at-hand and add the page to the
257  relevant free-list
258  */
259 
260  /*
261  3. Do step-2 repeatedly until no merging is possible
262  a. The buddy is not free
263  b. The max order is reached
264  */
265  do
266  {
267  page_t *const current_page = find_buddy(address,order);
268 
269  if( NULL == current_page || BUDDY_ADDR(address, order) != current_page->address )
270  {
271  CLEAR(g_pages[index].order);
272  list_add( &g_pages[index].list, &free_area[order] );
273  return;
274  }
275 
276  if( address > current_page->address )
277  {
278  address = current_page->address;
279  index = ADDR_TO_PAGE( address );
280  }
281 
282  list_del( &current_page->list );
283 
284  } while( order++ < MAX_ORDER );
285 
286 }
